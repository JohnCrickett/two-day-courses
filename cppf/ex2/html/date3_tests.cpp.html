1 #include "date3.hpp"
    2 #include <cassert>
    3 #include <stdexcept>
    4 
    5 int main(void)

    7     // checks valid and invalid dates

    9         try { mylib::date d(1971,9,13); } catch(std::out_of_range &) { assert(0); }
   10         try { mylib::date d(1900,1,1); } catch(std::out_of_range &) { assert(0); }
   11         try { mylib::date d(2099,12,31); } catch(std::out_of_range &) { assert(0); }
   12         try { mylib::date d(2000,2,29); } catch(std::out_of_range &) { assert(0); }
   13 
   14         try { mylib::date d(1971,0,13); assert(0); } catch(std::out_of_range &) {}
   15         try { mylib::date d(1971,9,0); assert(0); } catch(std::out_of_range &) {}
   16         try { mylib::date d(1971,13,9); assert(0); } catch(std::out_of_range &) {}

   18     }
   19 
   20     // check offsets around a leap year day

   22         mylib::date d1(2000,2,29);
   23         mylib::date d2(2000,3,1);
   24         assert( d2 - d1 == 1 );
   25         assert( !(d2 == d1) );

   27         assert( d2 == d1 );
   28         d1 += -10;
   29         d2 += 10;
   30         assert( !(d1 == d2) );
   31         assert( d2 - d1 == 20 );
   32     }
   33 
   34     // check that days between 1900/1/1 and 2099/12/31 are correct
   35     {
   36         mylib::date d1(1900,1,1);
   37         mylib::date d2(2099,12,31);
   38         assert( d2 - d1 == 73048 );
