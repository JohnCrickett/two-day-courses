    3 static bool is_leap_year(int year)
    4 {
    5     return (year % 400 == 0) || ((year % 4 == 0) && (year % 100 != 0));
    6 }

    8 static int days_in_month(int year, int month)
    9 {
   10     switch (month) {
   11     case 2:
   12         return is_leap_year(year) ? 29 : 28;
   13     case 4: case 6: case 9: case 11:
   14         return 30;
   15     }
   16     return 31;
   17 }

   19 static int to_day_number(const mylib_date * date)
   20 {
   21     int a = (14-date->month)/12;
   22     int y = date->year + 4800 - a;
   23     int m = date->month + 12*a - 3;
   24     int d = date->day + ((153*m+2)/5) + 365*y + (y/4) - (y/100) + (y/400) - 32045;
   25     return d;
   26 }

   28 static mylib_date to_ymd(int day_number)
   29 {
   30     int a = day_number + 32044;
   31     int b = (4*a + 3)/146097;
   32     int c = a-((146097*b)/4);
   33     int d = (4*c + 3)/1461;
   34     int e = c - (1461*d)/4;
   35     int m = (5*e + 2)/153;
   36     mylib_date date;
   37     date.year = 100*b + d - 4800 + (m/10);
   38     date.month = m + 3 - 12 * (m/10);
   39     date.day = e - ((153*m + 2)/5) + 1;
   40     return date;
   41 }

   43 bool mylib_date_is_valid(const mylib_date * date)
   44 {
   45     return (date->month >= 1 && date->month <= 12) &&
   46         (date->day >= 1 && date->day <= days_in_month(date->year, date->month));
   47 }

   51     mylib_date d = to_ymd(to_day_number(date) + offset_in_days);
