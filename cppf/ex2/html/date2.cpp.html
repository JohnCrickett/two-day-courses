    4 bool mylib::date::is_valid() const
    5 {
    6     return this->month >= 1 && this->month <= 12
    7         && this->day >= 1
    8         && this->day <= xyzlib_days_in_month(this->year, this->month);
    9 }

   11 void mylib::date::offset(int days)
   12 {
   13     int day_number = xyzlib_to_day_number(this->year, this->month, this->day);
   14     int new_day_number = day_number + days;
   15     xyzlib_to_ymd(new_day_number, &this->year, &this->month, &this->day);
   16 }

   18 int mylib::date::diff(const mylib::date & other) const
   19 {
   20     int day_number = xyzlib_to_day_number(this->year, this->month, this->day);
   21     int other_day_number = xyzlib_to_day_number(
   22         other.year, other.month, other.day);
