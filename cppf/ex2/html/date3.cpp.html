   10     int days_in_month(int year, int month)
   11     {
   12         switch (month) {
   13         case 2:
   14             return is_leap_year(year) ? 29 : 28;
   15         case 4: case 6: case 9: case 11:
   16             return 30;
   17         }
   18         return 31;
   19     }

   21     int to_day_number(const mylib::date & date)
   22     {
   23         int a = (14-date.month())/12;
   24         int y = date.year() + 4800 - a;
   25         int m = date.month() + 12*a - 3;
   26         int d = date.day() + ((153*m+2)/5) + 365*y
   27             + (y/4) - (y/100) + (y/400) - 32045;
   28         return d;
   29     }

   31     mylib::date to_date(int day_number)
   32     {
   33         int a = day_number + 32044;
   34         int b = (4*a + 3)/146097;
   35         int c = a-((146097*b)/4);
   36         int d = (4*c + 3)/1461;
   37         int e = c - (1461*d)/4;
   38         int m = (5*e + 2)/153;
   39         int year = 100*b + d - 4800 + (m/10);
   40         int month = m + 3 - 12 * (m/10);
   41         int day = e - ((153*m + 2)/5) + 1;
   42         return mylib::date(year,month,day);
   43     }
   44 }
