    4 static void bubble_sort_dates(mylib::date * base, size_t number_of_elements)
    5 {
    6     for(;;) {
    7         bool swapped = false;
    8         for (size_t i = 0; i < number_of_elements - 1; i++) 
    9             if (base[i].diff(base[i+1]) > 0) {
   10                 mylib::date tmp = base[i];
   11                 base[i] = base[i+1];
   12                 base[i+1] = tmp;
   13                 swapped = true;
   14             }
   15         if (!swapped)
   16             return;
   17     }
   18 }
