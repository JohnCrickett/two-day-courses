    5 static mylib_date * to_date(int year, int month, int day) {
    6     mylib_date d2 = {year,month,day};
    7     static mylib_date d;
    8     d = d2;
    9     return &d;
   10 }
   11 
   12 static void bubble_sort_dates(mylib_date * base, size_t number_of_elements)
   13 {
   14     for(;;) {
   15         bool swapped = false;
   16         for (size_t i = 0; i < number_of_elements - 1; i++) 
   17             if (mylib_date_diff(&base[i],&base[i+1]) > 0) {
   18                 mylib_date tmp = base[i];
   19                 base[i] = base[i+1];
   20                 base[i+1] = tmp;
   21                 swapped = true;
   22             }
   23         if (!swapped)
   24             return;
   25     }
   26 }

   28 int main(void)
   29 {
   30     // checks valid and invalid dates
   31     {
   32         assert( mylib_date_is_valid(to_date(1971,9,13)) );
   33         assert( mylib_date_is_valid(to_date(1900,1,1)) );
   34         assert( mylib_date_is_valid(to_date(2099,12,31)) );
   35         assert( mylib_date_is_valid(to_date(2000,2,29)) );
   36         assert( !mylib_date_is_valid(to_date(1971,0,13)) );
   37         assert( !mylib_date_is_valid(to_date(1971,9,0)) );
   38         assert( !mylib_date_is_valid(to_date(1971,13,9)) );
   39         assert( !mylib_date_is_valid(to_date(1971,9,31)) );
   40     }
   41 
   42     // check offsets around a leap year day
   43     {
   44         mylib_date d1 = {2000,2,29};
   45         mylib_date d2 = {2000,3,1};
   46         assert( mylib_date_diff(&d2,&d1) == 1 );
   47         assert( !mylib_date_is_equal(&d1,&d2) );
   48         mylib_date_offset(&d1,1);
   49         assert( mylib_date_diff(&d2,&d1) == 0 );
   50         assert( mylib_date_is_equal(&d1,&d2) );
   51         mylib_date_offset(&d1,-10);
   52         mylib_date_offset(&d2,10);
   53         assert( !mylib_date_is_equal(&d1,&d2) );
   54         assert( mylib_date_diff(&d2,&d1) == 20 );
   55     }
   56 
   57     // check that all dates between 1900/1/1 and 2099/12/31 are valid
   58     {
   59         mylib_date d1 = {1900,1,1};
   60         mylib_date d2 = {2099,12,31};
   61         assert( mylib_date_diff(&d2,&d1) == 73048 );
   62         assert( mylib_date_diff(&d1,&d2) == -73048 );
   63         for (mylib_date d = d1; !mylib_date_is_equal(&d,&d2); mylib_date_offset(&d,1))
   64             assert(mylib_date_is_valid(&d));
   65         for (mylib_date d = d2; !mylib_date_is_equal(&d,&d1); mylib_date_offset(&d,-1))
   66             assert(mylib_date_is_valid(&d));
   67     }
   68 
   69     // check that we are able to bubble sort an array of dates
   70     {
   71         mylib_date dates[] = {{2002,3,13}, {2005,4,20}, {1971,9,13}, {1970,5,1}};
   72         bubble_sort_dates(dates, 4);
   73         assert( mylib_date_is_equal(&dates[0],to_date(1970,5,1)) );
   74         assert( mylib_date_is_equal(&dates[1],to_date(1971,9,13)) );
   75         assert( mylib_date_is_equal(&dates[2],to_date(2002,3,13)) );
   76         assert( mylib_date_is_equal(&dates[3],to_date(2005,4,20)) );
   77     }
   78 
   79 }
