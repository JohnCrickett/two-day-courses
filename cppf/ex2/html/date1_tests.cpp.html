1 #include "date1.hpp"
    2 #include <assert.h>
    3 
    4 static mylib_date * to_date(int year, int month, int day) {
    5     mylib_date d2 = {year,month,day};
    6     static mylib_date d;
    7     d = d2;
    8     return &d;
    9 }
   10 
   11 int main(void)
   12 {
   13     // checks valid and invalid dates
   14     {
   15         assert( mylib_date_is_valid(to_date(1971,9,13)) );
   16         assert( mylib_date_is_valid(to_date(1900,1,1)) );
   17         assert( mylib_date_is_valid(to_date(2099,12,31)) );
   18         assert( mylib_date_is_valid(to_date(2000,2,29)) );
   19         assert( !mylib_date_is_valid(to_date(1971,0,13)) );
   20         assert( !mylib_date_is_valid(to_date(1971,9,0)) );
   21         assert( !mylib_date_is_valid(to_date(1971,13,9)) );
   22         assert( !mylib_date_is_valid(to_date(1971,9,31)) );
   23     }
   24 
   25     // check offsets around a leap year day
   26     {
   27         mylib_date d1 = {2000,2,29};
   28         mylib_date d2 = {2000,3,1};
   29         assert( mylib_date_diff(&d2,&d1) == 1 );
   30         assert( !mylib_date_is_equal(&d1,&d2) );
   31         mylib_date_offset(&d1,1);
   32         assert( mylib_date_diff(&d2,&d1) == 0 );
   33         assert( mylib_date_is_equal(&d1,&d2) );
   34         mylib_date_offset(&d1,-10);
   35         mylib_date_offset(&d2,10);
   36         assert( !mylib_date_is_equal(&d1,&d2) );
   37         assert( mylib_date_diff(&d2,&d1) == 20 );
   38     }
   39 
   40     // check that all dates between 1900/1/1 and 2099/12/31 are valid
   41     {
   42         mylib_date d1 = {1900,1,1};
   43         mylib_date d2 = {2099,12,31};
   44         assert( mylib_date_diff(&d2,&d1) == 73048 );
   45         assert( mylib_date_diff(&d1,&d2) == -73048 );
   46         for (mylib_date d = d1; !mylib_date_is_equal(&d,&d2); mylib_date_offset(&d,1))
   47             assert(mylib_date_is_valid(&d));
   48         for (mylib_date d = d2; !mylib_date_is_equal(&d,&d1); mylib_date_offset(&d,-1))
   49             assert(mylib_date_is_valid(&d));
   50     }
   51 }
