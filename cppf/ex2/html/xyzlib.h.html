    4 static bool xyzlib_is_leap_year(int year)
    5 {
    6     return (year % 400 == 0) || ((year % 4 == 0) && (year % 100 != 0));
    7 }
    8 
    9 static int xyzlib_days_in_month(int year, int month)
   10 {
   11     switch (month) {
   12     case 2:
   13         return xyzlib_is_leap_year(year) ? 29 : 28;
   14     case 4: case 6: case 9: case 11:
   15         return 30;
   16     }
   17     return 31;
   18 }

   20 static int xyzlib_to_day_number(int year, int month, int day)
   21 {
   22     int a = (14-month)/12;
   23     int y = year + 4800 - a;
   24     int m = month + 12*a - 3;
   25     int d = day + ((153*m+2)/5) + 365*y + (y/4) - (y/100) + (y/400) - 32045;
   26     return d;
   27 }

   29 static void xyzlib_to_ymd(int day_number, int * year, int * month, int * day)
   30 {
   31     int a = day_number + 32044;
   32     int b = (4*a + 3)/146097;
   33     int c = a-((146097*b)/4);
   34     int d = (4*c + 3)/1461;
   35     int e = c - (1461*d)/4;
   36     int m = (5*e + 2)/153;
   37     *year = 100*b + d - 4800 + (m/10);
   38     *month = m + 3 - 12 * (m/10);
   39     *day = e - ((153*m + 2)/5) + 1;
   40 }
