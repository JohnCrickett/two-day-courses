1 #include "directory.hpp"

    4 #include <string>

    7 class my_directory_printer : public mylib::directory_entry_visitor
    8 {
    9 public:
   10     explicit my_directory_printer(int indent_size = 4) : indent_size(indent_size), indent() {}
   11     void visit(mylib::directory & entry);
   12     void leave(mylib::directory & entry);
   13     void visit(mylib::file & entry);
   14 private:
   15     int indent_size;
   16     int indent;
   17 };
   18 
   19 void my_directory_printer::visit(mylib::directory & entry)
   20 {
   21     const std::string prefix(indent, ' ');
   22     std::cout << prefix << entry.name() << "/" << std::endl;
   23     indent += indent_size;
   24 }
   25 
   26 void my_directory_printer::leave(mylib::directory &)
   27 {
   28     indent -= indent_size;
   29 }
   30 
   31 void my_directory_printer::visit(mylib::file & entry)
   32 {
   33     const std::string prefix(indent, ' ');
   34     std::cout << prefix << entry.name() << std::endl;
   35 }

   44     mylib::file boo("boo");

   46

   51     gaz.add(boo);
